Declare Variables let, var, const

Declare three variables: name, age, and isStudent with appropriate values.

// Declaring variables with different data types let name = "Alex Johnson"; let age = 25; let isStudent = true; // Display the variables console.log("Name:", name); console.log("Age:", age); console.log("Is Student:", isStudent);

Calculate Total Arithmetic

Declare numeric variables and calculate their sum.

// Declaring numeric variables let num1 = 15; let num2 = 27; // Calculate the sum let sum = num1 + num2; // Display the result console.log(`${num1} + ${num2} = ${sum}`); console.log("Sum:", sum);

String Concatenation Strings

Combine first and last names using string concatenation.

// Declaring string variables let firstName = "Sarah"; let lastName = "Williams"; // String concatenation methods let fullName1 = firstName + " " + lastName; // Traditional let fullName2 = `${firstName} ${lastName}`; // Template literals console.log("Method 1:", fullName1); console.log("Method 2:", fullName2);

Constants const

Declare a constant and observe what happens when trying to reassign it.

// Declaring a constant const PI = 3.14159; console.log("PI value:", PI); // Attempting to reassign (this will cause an error) try { PI = 3.14; // This will throw an error } catch (error) { console.log("Error:", error.message); }

Working with Arrays Arrays

Create and manipulate arrays with fruit names.

// Declaring an array let fruits = ["apple", "banana", "orange"]; // Access second element (index 1) console.log("Second fruit:", fruits[1]); // Add a new fruit fruits.push("grape"); // Display updated array console.log("Updated fruits array:", fruits); console.log("Array length:", fruits.length);
7

Object Variables Objects

Create and access object properties using dot notation.

// Creating an object let person = { name: "Emma Thompson", age: 28, city: "New York" }; // Accessing properties with dot notation console.log("Person's name:", person.name); console.log("Person's age:", person.age); // Accessing with bracket notation console.log("Person's city:", person["city"]); // Display entire object console.log("Complete person object:", person);
8

Variable Scope Scope

Understand global and local variable scope.

// Global variable let globalVar = "I'm global!"; // Function to demonstrate scope function demonstrateScope() { // Local variable let localVar = "I'm local!"; console.log("Inside function - Global:", globalVar); console.log("Inside function - Local:", localVar); } demonstrateScope(); console.log("Outside function - Global:", globalVar); // Try to access local variable outside (will cause error) try { console.log(localVar); } catch (error) { console.log("Error accessing localVar:", error.message); }
9

Using typeof Operator typeof

Check data types of different variables.

// Variables of different types let myString = "Hello World"; let myNumber = 42; let myBoolean = true; let myArray = [1, 2, 3]; let myObject = {key: "value"}; let myUndefined; // Check types console.log("typeof myString:", typeof myString); console.log("typeof myNumber:", typeof myNumber); console.log("typeof myBoolean:", typeof myBoolean); console.log("typeof myArray:", typeof myArray); console.log("typeof myObject:", typeof myObject); console.log("typeof myUndefined:", typeof myUndefined);
10

Changing Variable Values Increment

Initialize a counter and update its value.

// Initialize counter let counter = 5; console.log("Initial counter value:", counter); // Different ways to increment counter = counter + 1; // Method 1 console.log("After adding 1:", counter); counter += 1; // Method 2 console.log("After += 1:", counter); counter++; // Method 3 console.log("After counter++:", counter); console.log("Final counter value:", counter);
