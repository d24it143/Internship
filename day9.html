Day 09: JavaScript Function Tasks

1. Hello World Function

function sayHello() { console.log("Hello, World!"); } sayHello();

2. Sum Function

function sum(a, b) { return a + b; } console.log(sum(5, 7)); // 12

3. Fahrenheit to Celsius

function fahrenheitToCelsius(f) { return (f - 32) * 5 / 9; } console.log(fahrenheitToCelsius(100)); // 37.78

4. Circle Area

function circleArea(r) { return Math.PI * r * r; } console.log(circleArea(5)); // 78.54

5. Reverse String

function reverseString(str) { return str.split('').reverse().join(''); } console.log(reverseString("hello")); // "olleh"

6. Check Even Number

function isEven(num) { return num % 2 === 0; } console.log(isEven(4)); // true console.log(isEven(5)); // false

7. Multiply Array Elements

function multiplyArray(arr) { return arr.reduce((prod, n) => prod * n, 1); } console.log(multiplyArray([1, 2, 3, 4])); // 24

8. Get Even Numbers from Array

function getEvens(arr) { return arr.filter(n => n % 2 === 0); } console.log(getEvens([1, 2, 3, 4, 5, 6])); // [2, 4, 6]

9. Find Largest Number in Array

function findMax(arr) { return Math.max(...arr); } console.log(findMax([10, 20, 5, 40])); // 40

10. Count Vowels in a String

function countVowels(str) { const vowels = 'aeiouAEIOU'; return [...str].filter(c => vowels.includes(c)).length; } console.log(countVowels("JavaScript")); // 3
